// Fungsi doPost utama untuk menerima request dari GitHub Pages / Frontend via fetch()
function doPost(e) {
  // Pengaturan dasar CORS
  const headers = {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "POST, GET, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type"
  };


  try {
    const requestData = JSON.parse(e.postData.contents);
    const action = requestData.action;
    const args = requestData.args || [];
    let result = {};


    // 1. Tarik Data Global
    if (action === 'getAppData') result = getAppData();
   
    // 2. Autentikasi Login
    else if (action === 'loginUser') result = loginUser(args[0], args[1]);
   
    // 3. Transaksi & Kalkulasi Stock
    else if (action === 'saveTransaction') result = saveTransaction(args[0], args[1]);
   
    // 4. Data Adjustment
    else if (action === 'saveAdjustment') result = saveAdjustment(args[0]);
    else if (action === 'approveAdjustment') result = approveAdjustment(args[0]);
    else if (action === 'rejectAdjustment') result = rejectAdjustment(args[0]);
    else if (action === 'sendAdjustmentEmail') result = sendAdjustmentEmail(args[0]);
   
    // 5. Edit/Hapus Stock Manual
    else if (action === 'updateStockData') result = updateStockData(args[0]);
    else if (action === 'deleteStockData') result = deleteStockData(args[0]);


    // 6. ROUTING PENGATURAN & MASTER DATA
    else if (action === 'saveSettingsData') result = saveSettingsData(args[0]);
    else if (action === 'addResto') result = addResto(args[0]);
    else if (action === 'deleteResto') result = deleteResto(args[0]);
    else if (action === 'addCategory') result = addCategory(args[0]);
    else if (action === 'deleteCategory') result = deleteCategory(args[0]);
    else if (action === 'saveUser') result = saveUser(args[0]);
    else if (action === 'deleteUser') result = deleteUser(args[0]);
    else if (action === 'saveMaster') result = saveMaster(args[0]);
    else if (action === 'deleteMaster') result = deleteMaster(args[0]);


    // 7. SURAT JALAN & WMS LOGIC
    else if (action === 'saveSuratJalanData') result = saveSuratJalanData(args[0]);
    else if (action === 'receiveSuratJalanData') result = receiveSuratJalanData(args[0]);
   
    // Jika tidak ada perintah yang cocok
    else throw new Error("Aksi backend tidak dikenali: " + action);


    // Kirim kembalian sukses ke Frontend
    return ContentService.createTextOutput(JSON.stringify({ success: true, data: result }))
      .setMimeType(ContentService.MimeType.JSON);


  } catch (error) {
    // Tangkap error dan kembalikan pesan
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function initializeDatabase() {

      sheet = ss.insertSheet(name);

  };


  setupSheet(SHEETS.USERS, ['ID', 'Nama', 'Username', 'Password', 'Role', 'Resto']);

function getAppData() {
  // Parsing khusus untuk Surat Jalan JSON
  const rawSJ = getTableData(SHEETS.SURAT_JALAN);
  const parsedSJ = rawSJ.map(row => {
    try {
      let parsed = JSON.parse(row.Data_JSON || row.data_json);
      parsed.status = row.Status || row.status; // pastikan status dari kolom tersinkronisasi
      return parsed;
    } catch(e) {
      return null;
    }
  }).filter(item => item !== null);


  return {
    settings: getSheetDataAsObject(SHEETS.SETTINGS, 'Key', 'Value'),
    restos: getSheetDataAsArray(SHEETS.RESTO),
    categories: getSheetDataAsArray(SHEETS.CATEGORY),
    users: getTableData(SHEETS.USERS),
    master: getTableData(SHEETS.MASTER),
    stock: getTableData(SHEETS.STOCK), // PENTING: getTableData otomatis memakai getDisplayValues
    inbound: getTableData(SHEETS.INBOUND),
    outbound: getTableData(SHEETS.OUTBOUND),
    adjustments: getTableData(SHEETS.ADJUSTMENT),
    suratJalan: parsedSJ // Data WMS
  };
}


// 2. AUTHENTICATION
function loginUser(username, password) {
  const users = getTableData(SHEETS.USERS);
  const user = users.find(u => String(u.Username).trim() === String(username).trim() && u.Password === password);
  if (user) {
    return { success: true, user: { id: user.ID, name: user.Nama, username: user.Username, role: user.Role, resto: user.Resto } };
  }
  return { success: false };
}


// 3. TRANSACTIONS & STOCK CALCULATION REALTIME SINKRONISASI
function saveTransaction(type, data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetName = type === 'inbound' ? SHEETS.INBOUND : SHEETS.OUTBOUND;
  const trxSheet = ss.getSheetByName(sheetName);
 
  let timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm");
  if (data.tanggal) {
      timestamp = String(data.tanggal).replace('T', ' ');
  }
  const id = (type === 'inbound' ? 'IN-' : 'OUT-') + Date.now() + Math.floor(Math.random() * 1000); // Mencegah duplikasi ID cepat
 
  // Pembersihan karakter untuk scanner
  const cleanKode = String(data.kode || '').trim().toUpperCase();
  const cleanResto = String(data.resto || '').trim().toUpperCase();


  // MENGUBAH qty menjadi absolut agar nilai (-) dari frontend untuk outbound tetap log bersih sbg bilangan positif di Sheet Riwayat
  const parsedQty = parseFloat(data.qty);
  const logQty = isNaN(parsedQty) ? data.qty : Math.abs(parsedQty);


  trxSheet.appendRow([
    id, timestamp, cleanKode, String(data.nama).trim(), data.kategori, logQty, data.satuan, data.petugas, cleanResto
  ]);


  const stockSheet = ss.getSheetByName(SHEETS.STOCK);
  const stockData = stockSheet.getDataRange().getDisplayValues(); // Gunakan Display Values
 
  if (stockData.length > 0) {
      const headers = stockData[0].map(h => String(h).replace(/\s+/g, '').toLowerCase());
     
      const cKode = headers.indexOf('kodebahan');
      const cNama = headers.indexOf('namabahan');
      const cKat = headers.indexOf('kategori');
      const cResto = headers.indexOf('resto') > -1 ? headers.indexOf('resto') : headers.indexOf('lokasi');
      const cQtyB = headers.indexOf('qtybesar') > -1 ? headers.indexOf('qtybesar') : headers.indexOf('stockbesar');
      const cQtyK = headers.indexOf('qtykecil') > -1 ? headers.indexOf('qtykecil') : headers.indexOf('stockkecil');
   
      const iKode = cKode > -1 ? cKode : 0;
      const iNama = cNama > -1 ? cNama : 1;
      const iKat = cKat > -1 ? cKat : 2;
      const iResto = cResto > -1 ? cResto : 3;
      const iQtyB = cQtyB > -1 ? cQtyB : 4;
      const iQtyK = cQtyK > -1 ? cQtyK : 5;
   
      let rowIndex = -1;
   
      for (let i = 1; i < stockData.length; i++) {
        let rowKode = String(stockData[i][iKode] || '').trim().toUpperCase();
        let rowResto = String(stockData[i][iResto] || '').trim().toUpperCase();
   
        if (rowKode === cleanKode && rowResto === cleanResto) {
            rowIndex = i + 1;
            break;
        }
      }
   
      if (data.isDepleted && rowIndex > -1) {
        stockSheet.deleteRow(rowIndex);
      }
      else if (data.isNewStock && rowIndex === -1) {
        let newRow = new Array(stockData[0].length).fill('');
        newRow[iKode] = cleanKode;
        newRow[iNama] = String(data.nama).trim();
        newRow[iKat] = data.kategori;
        newRow[iResto] = cleanResto;
        newRow[iQtyB] = data.newQtyBesar;
        newRow[iQtyK] = data.newQtyKecil;
        stockSheet.appendRow(newRow);
      }
      else if (rowIndex > -1) {
        stockSheet.getRange(rowIndex, iQtyB + 1).setValue(data.newQtyBesar);
        stockSheet.getRange(rowIndex, iQtyK + 1).setValue(data.newQtyKecil);
      }
  }


  SpreadsheetApp.flush(); // Memastikan penulisan ke sheet selesai
  return getAppData();
}


// 4. SURAT JALAN / PENGIRIMAN BARANG (WMS LOGIC)
function saveSuratJalanData(payloadSj) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEETS.SURAT_JALAN);
  if (!sheet) {
    sheet = ss.insertSheet(SHEETS.SURAT_JALAN);
    sheet.appendRow(['ID', 'Data_JSON', 'Status']); // Buat sheet jika belum ada
  }
 
  // Simpan Payload Surat Jalan dalam format stringified JSON agar items tersimpan dalam satu kolom
  sheet.appendRow([payloadSj.id, JSON.stringify(payloadSj), payloadSj.status]);
 
  // Memotong Stok (Barang Keluar) dari Gudang Pusat / Pengirim
  payloadSj.items.forEach(item => {
    const outPayload = {
        kode: item.kode, nama: item.nama, kategori: item.kategori, qty: "-" + item.qty, satuan: item.satuan,
        petugas: payloadSj.pengirim, resto: 'Resto Pusat', // Asumsi pengirim adalah gudang utama. Bisa disesuaikan jika dinamis
       
        // Data ini dikalkulasi oleh backend (karena dari frontend WMS tidak membawa perhitungan konversi)
        isDepleted: false,
        isNewStock: false
    };


    // Melakukan simulasi perhitungan sisa stock khusus untuk pemotongan massal SJ
    const masterData = getTableData(SHEETS.MASTER);
    const mItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === String(item.kode).trim().toUpperCase());
   
    if(mItem) {
        const stockData = ss.getSheetByName(SHEETS.STOCK).getDataRange().getDisplayValues();
        let qtyKecilInput = Math.abs(parseFloat(item.qty));
       
        if(String(item.satuan).toUpperCase() === String(mItem.SatuanBesar).toUpperCase()) {
            qtyKecilInput = qtyKecilInput * parseFloat(mItem.Konversi);
        }
       
        let existingIndex = -1;
        for(let i=1; i<stockData.length; i++) {
           if(stockData[i][0] === item.kode && stockData[i][3] === 'Resto Pusat') {
               existingIndex = i; break;
           }
        }
       
        if(existingIndex > -1) {
            let currentQty = parseFloat(String(stockData[existingIndex][5]).replace(/[^0-9.-]+/g,"")) || 0;
            let newQtyKecil = currentQty - qtyKecilInput;
            outPayload.isDepleted = newQtyKecil <= 0;
            outPayload.newQtyKecil = newQtyKecil + ' ' + mItem.SatuanKecil;
            outPayload.newQtyBesar = (newQtyKecil / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        }
    }
   
    saveTransaction('outbound', outPayload);
  });


  SpreadsheetApp.flush();
  return getAppData();
}


function receiveSuratJalanData(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEETS.SURAT_JALAN);
  if(sheet) {
    const sheetData = sheet.getDataRange().getValues();
    for (let i = 1; i < sheetData.length; i++) {
      if (sheetData[i][0] === data.id) {
        sheet.getRange(i + 1, 3).setValue('Diterima'); // Ubah status jadi Diterima
       
        // Perbarui isi JSON agar tersinkronisasi
        try {
            let parsed = JSON.parse(sheetData[i][1]);
            parsed.status = 'Diterima';
            sheet.getRange(i + 1, 2).setValue(JSON.stringify(parsed));
        } catch(e) {}
        break;
      }
    }
  }


  // Masukkan Stok ke Inbound (Barang Masuk) di Store Penerima
  data.items.forEach(item => {
    const masterData = getTableData(SHEETS.MASTER);
    const mItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === String(item.kode).trim().toUpperCase());
   
    if(mItem) {
        let qtyKecilInput = Math.abs(parseFloat(item.qty));
        if(String(item.satuan).toUpperCase() === String(mItem.SatuanBesar).toUpperCase()) {
            qtyKecilInput = qtyKecilInput * parseFloat(mItem.Konversi);
        }


        const stockData = ss.getSheetByName(SHEETS.STOCK).getDataRange().getDisplayValues();
        let existingIndex = -1;
        for(let i=1; i<stockData.length; i++) {
           if(stockData[i][0] === item.kode && stockData[i][3] === data.resto) {
               existingIndex = i; break;
           }
        }
       
        item.isNewStock = existingIndex === -1;
        if(existingIndex > -1) {
            let currentQty = parseFloat(String(stockData[existingIndex][5]).replace(/[^0-9.-]+/g,"")) || 0;
            let newQtyKecil = currentQty + qtyKecilInput;
            item.newQtyKecil = newQtyKecil + ' ' + mItem.SatuanKecil;
            item.newQtyBesar = (newQtyKecil / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        } else {
            item.newQtyKecil = qtyKecilInput + ' ' + mItem.SatuanKecil;
            item.newQtyBesar = (qtyKecilInput / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        }
    }
   
    saveTransaction('inbound', item);
  });


  SpreadsheetApp.flush();
  return getAppData();
}

function approveAdjustment(id) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const adjSheet = ss.getSheetByName(SHEETS.ADJUSTMENT);
  const rawData = adjSheet.getDataRange().getValues();
  const displayData = adjSheet.getDataRange().getDisplayValues(); // PENTING: Untuk menghindari bug Date Object


  for (let i = 1; i < rawData.length; i++) {
    // Status ada di index 8 (berdasarkan Array Header di initializeDatabase)
    if (rawData[i][0] == id && rawData[i][8] == 'Pending') {
      adjSheet.getRange(i + 1, 9).setValue('Approved');
     
      const kode = displayData[i][2];
      const qtyFisik = parseFloat(displayData[i][5]);
      const resto = displayData[i][10];
     
      forceUpdateStockForAdjustment(kode, resto, qtyFisik);
      break;
    }
  }
 
  SpreadsheetApp.flush(); // Memastikan penulisan ke sheet selesai
  return getAppData();
}


function rejectAdjustment(id) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const adjSheet = ss.getSheetByName(SHEETS.ADJUSTMENT);
  const data = adjSheet.getDataRange().getValues();


  for (let i = 1; i < data.length; i++) {
    // Status ada di index 8
    if (data[i][0] == id && data[i][8] == 'Pending') {
      adjSheet.getRange(i + 1, 9).setValue('Rejected');
      break;
    }
  }
 
  SpreadsheetApp.flush(); // Memastikan penulisan ke sheet selesai
  return getAppData();
}


function forceUpdateStockForAdjustment(kode, resto, qtyFisikBesar) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const stockSheet = ss.getSheetByName(SHEETS.STOCK);
  const stockData = stockSheet.getDataRange().getDisplayValues(); // PENTING: Text Based Read
  const masterData = getTableData(SHEETS.MASTER);
 
  if (stockData.length === 0) return;
 
  const headers = stockData[0].map(h => String(h).replace(/\s+/g, '').toLowerCase());
  const cKode = headers.indexOf('kodebahan') > -1 ? headers.indexOf('kodebahan') : 0;
  const cNama = headers.indexOf('namabahan') > -1 ? headers.indexOf('namabahan') : 1;
  const cKat = headers.indexOf('kategori') > -1 ? headers.indexOf('kategori') : 2;
  const cResto = headers.indexOf('resto') > -1 ? headers.indexOf('resto') : (headers.indexOf('lokasi') > -1 ? headers.indexOf('lokasi') : 3);
  const cQtyB = headers.indexOf('qtybesar') > -1 ? headers.indexOf('qtybesar') : (headers.indexOf('stockbesar') > -1 ? headers.indexOf('stockbesar') : 4);
  const cQtyK = headers.indexOf('qtykecil') > -1 ? headers.indexOf('qtykecil') : (headers.indexOf('stockkecil') > -1 ? headers.indexOf('stockkecil') : 5);


  const cleanKode = String(kode).trim().toUpperCase();
  const masterItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === cleanKode);
  if(!masterItem) return;


  let newQtyKecil = qtyFisikBesar * parseFloat(masterItem.Konversi);
  let newQtyBesarStr = qtyFisikBesar + ' ' + masterItem.SatuanBesar;
  let newQtyKecilStr = newQtyKecil + ' ' + masterItem.SatuanKecil;


  for (let i = 1; i < stockData.length; i++) {
    if (
      String(stockData[i][cKode] || '').trim().toUpperCase() === cleanKode &&
      String(stockData[i][cResto] || '').trim().toUpperCase() === String(resto).trim().toUpperCase()
    ) {
      if (newQtyKecil <= 0) {
          stockSheet.deleteRow(i + 1);
      } else {
          stockSheet.getRange(i + 1, cQtyB + 1).setValue(newQtyBesarStr);
          stockSheet.getRange(i + 1, cQtyK + 1).setValue(newQtyKecilStr);
      }
      return;
    }
  }
 
  if (newQtyKecil > 0) {
      let newRow = new Array(stockData[0].length).fill('');
      newRow[cKode] = cleanKode;
      newRow[cNama] = masterItem.NamaBahan;
      newRow[cKat] = masterItem.Kategori;
      newRow[cResto] = resto;
      newRow[cQtyB] = newQtyBesarStr;
      newRow[cQtyK] = newQtyKecilStr;
      stockSheet.appendRow(newRow);
  }
}

    // Safety check untuk mencegah error cell limit (>50,000 karakter) ketika save Base64 Logo
    if (key === 'logo_url' && typeof valueToSave === 'string' && valueToSave.length > 49000) {
        throw new Error("Kapasitas gambar terlalu besar. Mohon upload logo dengan dimensi/resolusi yang lebih kecil.");
    }


    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === key) {
        sheet.getRange(i + 1, 2).setValue(valueToSave);
        found = true;
        break;
      }
    }
    if (!found && appSet[key] !== undefined) {
      sheet.appendRow([key, valueToSave]);
    }
  });


  SpreadsheetApp.flush();
  return getAppData();
}

// --- UTILITY FUNCTIONS ---
function getTableData(sheetName) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  // PENTING: Gunakan getDisplayValues secara mutlak untuk konversi konsisten
  const data = sheet.getDataRange().getDisplayValues();
  const headers = data.shift();
  return data.map(row => {
    let obj = {};
    headers.forEach((header, index) => obj[header] = row[index]);
    return obj;
  });
}


function getSheetDataAsObject(sheetName, keyCol, valCol) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) return {};
  const data = sheet.getDataRange().getValues();
  let obj = {};
  for (let i = 1; i < data.length; i++) obj[data[i][0]] = data[i][1];
  return obj;
}


function getSheetDataAsArray(sheetName) {

  if (!sheet || sheet.getLastRow() < 2) return [];
  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues();
  return data.map(r => r[0]);
}

// Fungsi doPost utama untuk menerima request dari GitHub Pages / Frontend via fetch()
function doPost(e) {
  // Pengaturan dasar CORS
  const headers = {
    "Access-Control-Allow-Origin": "*",
    "Access-Control-Allow-Methods": "POST, GET, OPTIONS",
    "Access-Control-Allow-Headers": "Content-Type"
  };


  try {
    const requestData = JSON.parse(e.postData.contents);
    const action = requestData.action;
    const args = requestData.args || [];
    let result = {};


    // 1. Tarik Data Global
    if (action === 'getAppData') result = getAppData();
   
    // 2. Autentikasi Login
    else if (action === 'loginUser') result = loginUser(args[0], args[1]);
   
    // 3. Transaksi & Kalkulasi Stock
    else if (action === 'saveTransaction') result = saveTransaction(args[0], args[1]);
   
    // 4. Data Adjustment
    else if (action === 'saveAdjustment') result = saveAdjustment(args[0]);
    else if (action === 'approveAdjustment') result = approveAdjustment(args[0]);
    else if (action === 'rejectAdjustment') result = rejectAdjustment(args[0]);
    else if (action === 'sendAdjustmentEmail') result = sendAdjustmentEmail(args[0]);
   
    // 5. Edit/Hapus Stock Manual
    else if (action === 'updateStockData') result = updateStockData(args[0]);
    else if (action === 'deleteStockData') result = deleteStockData(args[0]);


    // 6. ROUTING PENGATURAN & MASTER DATA
    else if (action === 'saveSettingsData') result = saveSettingsData(args[0]);
    else if (action === 'addResto') result = addResto(args[0]);
    else if (action === 'deleteResto') result = deleteResto(args[0]);
    else if (action === 'addCategory') result = addCategory(args[0]);
    else if (action === 'deleteCategory') result = deleteCategory(args[0]);
    else if (action === 'saveUser') result = saveUser(args[0]);
    else if (action === 'deleteUser') result = deleteUser(args[0]);
    else if (action === 'saveMaster') result = saveMaster(args[0]);
    else if (action === 'deleteMaster') result = deleteMaster(args[0]);


    // 7. SURAT JALAN & WMS LOGIC
    else if (action === 'saveSuratJalanData') result = saveSuratJalanData(args[0]);
    else if (action === 'receiveSuratJalanData') result = receiveSuratJalanData(args[0]);
   
    // Jika tidak ada perintah yang cocok
    else throw new Error("Aksi backend tidak dikenali: " + action);


    // Kirim kembalian sukses ke Frontend
    return ContentService.createTextOutput(JSON.stringify({ success: true, data: result }))
      .setMimeType(ContentService.MimeType.JSON);


  } catch (error) {
    // Tangkap error dan kembalikan pesan
    return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// 3. TRANSACTIONS & STOCK CALCULATION REALTIME SINKRONISASI
function saveTransaction(type, data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetName = type === 'inbound' ? SHEETS.INBOUND : SHEETS.OUTBOUND;
  const trxSheet = ss.getSheetByName(sheetName);
 
  let timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm");
  if (data.tanggal) {
      timestamp = String(data.tanggal).replace('T', ' ');
  }
  const id = (type === 'inbound' ? 'IN-' : 'OUT-') + Date.now() + Math.floor(Math.random() * 1000); // Mencegah duplikasi ID cepat
 
  // Pembersihan karakter untuk scanner
  const cleanKode = String(data.kode || '').trim().toUpperCase();
  const cleanResto = String(data.resto || '').trim().toUpperCase();


  // MENGUBAH qty menjadi absolut agar nilai (-) dari frontend untuk outbound tetap log bersih sbg bilangan positif di Sheet Riwayat
  const parsedQty = parseFloat(data.qty);
  const logQty = isNaN(parsedQty) ? data.qty : Math.abs(parsedQty);


  trxSheet.appendRow([
    id, timestamp, cleanKode, String(data.nama).trim(), data.kategori, logQty, data.satuan, data.petugas, cleanResto
  ]);


  const stockSheet = ss.getSheetByName(SHEETS.STOCK);
  const stockData = stockSheet.getDataRange().getDisplayValues(); // Gunakan Display Values
 
  if (stockData.length > 0) {
      const headers = stockData[0].map(h => String(h).replace(/\s+/g, '').toLowerCase());
     
      const cKode = headers.indexOf('kodebahan');
      const cNama = headers.indexOf('namabahan');
      const cKat = headers.indexOf('kategori');
      const cResto = headers.indexOf('resto') > -1 ? headers.indexOf('resto') : headers.indexOf('lokasi');
      const cQtyB = headers.indexOf('qtybesar') > -1 ? headers.indexOf('qtybesar') : headers.indexOf('stockbesar');
      const cQtyK = headers.indexOf('qtykecil') > -1 ? headers.indexOf('qtykecil') : headers.indexOf('stockkecil');
   
      const iKode = cKode > -1 ? cKode : 0;
      const iNama = cNama > -1 ? cNama : 1;
      const iKat = cKat > -1 ? cKat : 2;
      const iResto = cResto > -1 ? cResto : 3;
      const iQtyB = cQtyB > -1 ? cQtyB : 4;
      const iQtyK = cQtyK > -1 ? cQtyK : 5;
   
      let rowIndex = -1;
   
      for (let i = 1; i < stockData.length; i++) {
        let rowKode = String(stockData[i][iKode] || '').trim().toUpperCase();
        let rowResto = String(stockData[i][iResto] || '').trim().toUpperCase();
   
        if (rowKode === cleanKode && rowResto === cleanResto) {
            rowIndex = i + 1;
            break;
        }
      }
   
      if (data.isDepleted && rowIndex > -1) {
        stockSheet.deleteRow(rowIndex);
      }
      else if (data.isNewStock && rowIndex === -1) {
        let newRow = new Array(stockData[0].length).fill('');
        newRow[iKode] = cleanKode;
        newRow[iNama] = String(data.nama).trim();
        newRow[iKat] = data.kategori;
        newRow[iResto] = cleanResto;
        newRow[iQtyB] = data.newQtyBesar;
        newRow[iQtyK] = data.newQtyKecil;
        stockSheet.appendRow(newRow);
      }
      else if (rowIndex > -1) {
        stockSheet.getRange(rowIndex, iQtyB + 1).setValue(data.newQtyBesar);
        stockSheet.getRange(rowIndex, iQtyK + 1).setValue(data.newQtyKecil);
      }
  }


  SpreadsheetApp.flush(); // Memastikan penulisan ke sheet selesai
  return getAppData();
}


// 4. SURAT JALAN / PENGIRIMAN BARANG (WMS LOGIC)
function saveSuratJalanData(payloadSj) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEETS.SURAT_JALAN);
  if (!sheet) {
    sheet = ss.insertSheet(SHEETS.SURAT_JALAN);
    sheet.appendRow(['ID', 'Data_JSON', 'Status']); // Buat sheet jika belum ada
  }
 
  // Simpan Payload Surat Jalan dalam format stringified JSON agar items tersimpan dalam satu kolom
  sheet.appendRow([payloadSj.id, JSON.stringify(payloadSj), payloadSj.status]);
 
  // Memotong Stok (Barang Keluar) dari Gudang Pusat / Pengirim
  payloadSj.items.forEach(item => {
    const outPayload = {
        kode: item.kode, nama: item.nama, kategori: item.kategori, qty: "-" + item.qty, satuan: item.satuan,
        petugas: payloadSj.pengirim, resto: 'Resto Pusat', // Asumsi pengirim adalah gudang utama. Bisa disesuaikan jika dinamis
       
        // Data ini dikalkulasi oleh backend (karena dari frontend WMS tidak membawa perhitungan konversi)
        isDepleted: false,
        isNewStock: false
    };


    // Melakukan simulasi perhitungan sisa stock khusus untuk pemotongan massal SJ
    const masterData = getTableData(SHEETS.MASTER);
    const mItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === String(item.kode).trim().toUpperCase());
   
    if(mItem) {
        const stockData = ss.getSheetByName(SHEETS.STOCK).getDataRange().getDisplayValues();
        let qtyKecilInput = Math.abs(parseFloat(item.qty));
       
        if(String(item.satuan).toUpperCase() === String(mItem.SatuanBesar).toUpperCase()) {
            qtyKecilInput = qtyKecilInput * parseFloat(mItem.Konversi);
        }
       
        let existingIndex = -1;
        for(let i=1; i<stockData.length; i++) {
           if(stockData[i][0] === item.kode && stockData[i][3] === 'Resto Pusat') {
               existingIndex = i; break;
           }
        }
       
        if(existingIndex > -1) {
            let currentQty = parseFloat(String(stockData[existingIndex][5]).replace(/[^0-9.-]+/g,"")) || 0;
            let newQtyKecil = currentQty - qtyKecilInput;
            outPayload.isDepleted = newQtyKecil <= 0;
            outPayload.newQtyKecil = newQtyKecil + ' ' + mItem.SatuanKecil;
            outPayload.newQtyBesar = (newQtyKecil / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        }
    }
   
    saveTransaction('outbound', outPayload);
  });


  SpreadsheetApp.flush();
  return getAppData();
}


function receiveSuratJalanData(data) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(SHEETS.SURAT_JALAN);
  if(sheet) {
    const sheetData = sheet.getDataRange().getValues();
    for (let i = 1; i < sheetData.length; i++) {
      if (sheetData[i][0] === data.id) {
        sheet.getRange(i + 1, 3).setValue('Diterima'); // Ubah status jadi Diterima
       
        // Perbarui isi JSON agar tersinkronisasi
        try {
            let parsed = JSON.parse(sheetData[i][1]);
            parsed.status = 'Diterima';
            sheet.getRange(i + 1, 2).setValue(JSON.stringify(parsed));
        } catch(e) {}
        break;
      }
    }
  }


  // Masukkan Stok ke Inbound (Barang Masuk) di Store Penerima
  data.items.forEach(item => {
    const masterData = getTableData(SHEETS.MASTER);
    const mItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === String(item.kode).trim().toUpperCase());
   
    if(mItem) {
        let qtyKecilInput = Math.abs(parseFloat(item.qty));
        if(String(item.satuan).toUpperCase() === String(mItem.SatuanBesar).toUpperCase()) {
            qtyKecilInput = qtyKecilInput * parseFloat(mItem.Konversi);
        }


        const stockData = ss.getSheetByName(SHEETS.STOCK).getDataRange().getDisplayValues();
        let existingIndex = -1;
        for(let i=1; i<stockData.length; i++) {
           if(stockData[i][0] === item.kode && stockData[i][3] === data.resto) {
               existingIndex = i; break;
           }
        }
       
        item.isNewStock = existingIndex === -1;
        if(existingIndex > -1) {
            let currentQty = parseFloat(String(stockData[existingIndex][5]).replace(/[^0-9.-]+/g,"")) || 0;
            let newQtyKecil = currentQty + qtyKecilInput;
            item.newQtyKecil = newQtyKecil + ' ' + mItem.SatuanKecil;
            item.newQtyBesar = (newQtyKecil / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        } else {
            item.newQtyKecil = qtyKecilInput + ' ' + mItem.SatuanKecil;
            item.newQtyBesar = (qtyKecilInput / parseFloat(mItem.Konversi)).toFixed(2) + ' ' + mItem.SatuanBesar;
        }
    }
   
    saveTransaction('inbound', item);
  });


  SpreadsheetApp.flush();
  return getAppData();
}

function approveAdjustment(id) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const rawData = adjSheet.getDataRange().getValues();
  const displayData = adjSheet.getDataRange().getDisplayValues(); // PENTING: Untuk menghindari bug Date Object


  for (let i = 1; i < rawData.length; i++) {
    // Status ada di index 8 (berdasarkan Array Header di initializeDatabase)
    if (rawData[i][0] == id && rawData[i][8] == 'Pending') {
      adjSheet.getRange(i + 1, 9).setValue('Approved');
     
      const kode = displayData[i][2];

      break;
    }
  }
 
  SpreadsheetApp.flush(); // Memastikan penulisan ke sheet selesai
  return getAppData();
}

function forceUpdateStockForAdjustment(kode, resto, qtyFisikBesar) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const stockSheet = ss.getSheetByName(SHEETS.STOCK);
  const stockData = stockSheet.getDataRange().getDisplayValues(); // PENTING: Text Based Read
  const masterData = getTableData(SHEETS.MASTER);
 
  if (stockData.length === 0) return;
 
  const headers = stockData[0].map(h => String(h).replace(/\s+/g, '').toLowerCase());
  const cKode = headers.indexOf('kodebahan') > -1 ? headers.indexOf('kodebahan') : 0;
  const cNama = headers.indexOf('namabahan') > -1 ? headers.indexOf('namabahan') : 1;
  const cKat = headers.indexOf('kategori') > -1 ? headers.indexOf('kategori') : 2;
  const cResto = headers.indexOf('resto') > -1 ? headers.indexOf('resto') : (headers.indexOf('lokasi') > -1 ? headers.indexOf('lokasi') : 3);
  const cQtyB = headers.indexOf('qtybesar') > -1 ? headers.indexOf('qtybesar') : (headers.indexOf('stockbesar') > -1 ? headers.indexOf('stockbesar') : 4);
  const cQtyK = headers.indexOf('qtykecil') > -1 ? headers.indexOf('qtykecil') : (headers.indexOf('stockkecil') > -1 ? headers.indexOf('stockkecil') : 5);


  const cleanKode = String(kode).trim().toUpperCase();
  const masterItem = masterData.find(m => String(m.KodeBahan).trim().toUpperCase() === cleanKode);
  if(!masterItem) return;


  let newQtyKecil = qtyFisikBesar * parseFloat(masterItem.Konversi);
  let newQtyBesarStr = qtyFisikBesar + ' ' + masterItem.SatuanBesar;
  let newQtyKecilStr = newQtyKecil + ' ' + masterItem.SatuanKecil;


  for (let i = 1; i < stockData.length; i++) {
    if (
      String(stockData[i][cKode] || '').trim().toUpperCase() === cleanKode &&
      String(stockData[i][cResto] || '').trim().toUpperCase() === String(resto).trim().toUpperCase()
    ) {
      if (newQtyKecil <= 0) {
          stockSheet.deleteRow(i + 1);
      } else {
          stockSheet.getRange(i + 1, cQtyB + 1).setValue(newQtyBesarStr);
          stockSheet.getRange(i + 1, cQtyK + 1).setValue(newQtyKecilStr);
      }
      return;
    }
  }
 
  if (newQtyKecil > 0) {
      let newRow = new Array(stockData[0].length).fill('');
      newRow[cKode] = cleanKode;
      newRow[cNama] = masterItem.NamaBahan;
      newRow[cKat] = masterItem.Kategori;
      newRow[cResto] = resto;
      newRow[cQtyB] = newQtyBesarStr;
      newRow[cQtyK] = newQtyKecilStr;
      stockSheet.appendRow(newRow);
  }
}

function saveSettingsData(appSet) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEETS.SETTINGS);
  const data = sheet.getDataRange().getValues();
 
  const keys = ['app_title', 'copyright', 'logo_url', 'admin_email', 'frontend_url'];
 
  keys.forEach(key => {
    let found = false;
    let valueToSave = appSet[key] !== undefined ? appSet[key] : '';
   
    // Safety check untuk mencegah error cell limit (>50,000 karakter) ketika save Base64 Logo
    if (key === 'logo_url' && typeof valueToSave === 'string' && valueToSave.length > 49000) {
        throw new Error("Kapasitas gambar terlalu besar. Mohon upload logo dengan dimensi/resolusi yang lebih kecil.");
    }


    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === key) {
        sheet.getRange(i + 1, 2).setValue(valueToSave);
        found = true;
        break;
      }
    }
    if (!found && appSet[key] !== undefined) {
      sheet.appendRow([key, valueToSave]);
    }
  });


  SpreadsheetApp.flush();
  return getAppData();
}

// --- UTILITY FUNCTIONS ---
function getTableData(sheetName) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  // PENTING: Gunakan getDisplayValues secara mutlak untuk konversi konsisten
  const data = sheet.getDataRange().getDisplayValues();
  const headers = data.shift();
  return data.map(row => {
    let obj = {};
    headers.forEach((header, index) => obj[header] = row[index]);
    return obj;
  });
}


function getSheetDataAsObject(sheetName, keyCol, valCol) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) return {};
  const data = sheet.getDataRange().getValues();
  let obj = {};
  for (let i = 1; i < data.length; i++) obj[data[i][0]] = data[i][1];
  return obj;
}


function getSheetDataAsArray(sheetName) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet || sheet.getLastRow() < 2) return [];
  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues();
  return data.map(r => r[0]);
}
